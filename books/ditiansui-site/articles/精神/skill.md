## 功能

在收到八字四柱后，先定义命局中的"精"（生我的印星）、"神"（克我的官杀或我生的食伤）、"气"（日主的本气是否贯足），然后评估三者之间的流通平衡状态，最后判断是否存在五行偏枯（水泛木浮、木多火炽等）以及给出损益调节建议。此技能在体用分析之后、源流判断之前调用。

## 输入

- **四柱八字**：年柱、月柱、日柱、时柱的天干地支
- **用神与喜忌**：已确定的用神和喜忌判断（可选，来自体用分析）
- **大运**：已排好的大运列表（可选，用于评估大运对精神气的影响）

## 处理逻辑

1. **精的定位与评估**
   a. 检查四柱中的正印和偏印（生扶日主的五行）
   b. 评估印星的力量：
   - 数量（透出几个印星）
   - 根气（印星是否通根于地支）
   - 是否为用神
     c. **依据**：精神篇【任氏曰】"精者，生我之神也"

2. **气的定位与评估**
   a. 评估日主的"本气贯足"程度：
   - 日主是否通根于地支
   - 日主在月令中的旺相休囚死状态
   - 日主的帮扶力量（比劫）
     b. 判断气是否"贯足"
     c. **依据**：精神篇【任氏曰】"气者，本气贯足也"

3. **神的定位与评估**
   a. 检查四柱中的正官七杀（克我的五行）和食神伤官（我生的五行）
   b. 识别主要的"神"（以官杀为内核的制约力量）
   c. 评估神的力量和状态
   d. **依据**：精神篇【任氏曰】"神者，克我之物也"

4. **精→气→神连锁分析**
   a. 检查精是否能够生助气（印星生扶日主）
   b. 检查气是否贯足（日主自身有力）
   c. 检查神是否有用（官杀制约或食伤泄秀）
   d. 综合评估链条完整性
   e. **依据**：精神篇【任氏曰】"精足则气旺，气旺则神旺"

5. **五行偏枯检测**
   a. 检查水泛木浮（水过旺、木被浮）
   b. 检查木多火炽（木过旺、火被塞）
   c. 检查火炎金无（火过旺、金被熔）
   d. 检查金多水弱（金过旺、水被泄）
   e. **依据**：精神篇【任氏曰】"水泛木浮，木无精神；木多火炽，火无精神..."

6. **损益建议**
   a. 精太足 → 建议益气（加强食伤泄秀）
   b. 气太旺 → 建议助神（加强官杀制约）
   c. 神太泄 → 建议滋精（加强印星生扶）
   d. **依据**：精神篇【任氏曰】"精太足宜益其气，气太旺宜助其神，神太泄宜滋其精"

interface JingShenReport {

  jing: {

    analysis: string

    dayMaster: string // 日干

    rootBranches: string[] // 通根地支
    analysis: string

    guanSha: string[] // 正官七杀列表
    shiShang: string[] // 食神伤官列表
    strength: '旺' | '中' | '弱'
    analysis: string
  }
  /** 连锁分析 */
  chain: {
    complete: boolean // 精→气→神链条是否完整
    blockage: string | null // 堵点描述
    assessment: string
  }
  /** 偏枯检测 */
  pianKu: {
    hasPianKu: boolean
    type: '水泛木浮' | '木多火炽' | '火炎金无' | '金多水弱' | null
    description: string
  }
  /** 损益建议 */
  advice: {
    action: '益精' | '益气' | '助神' | '泄精' | '抑气' | '滋神' | '平衡'
    reason: string
    preferredLuck: string[] // 推荐行运方向
  }
}
